Code comparison

Blinking a led in pure C

            
            #include <avr/io.h>
            #include <util/delay.h>

            #define BLINK_DELAY_MS 1000

            int main (void)
            {
             DDRB |= _BV(DDB5);

             while(1) {
              PORTB |= _BV(PORTB5);
              _delay_ms(BLINK_DELAY_MS);

              PORTB &= ~_BV(PORTB5);
              _delay_ms(BLINK_DELAY_MS);
             }
            }

Code comparison

Blinking a led in MicroPython

            
                import time
                import machine

                led = machine.Pin(15, machine.Pin.OUT)
                while True:
                    led.high()
                    time.sleep(0.5)
                    led.low()
                    time.sleep(0.5)
